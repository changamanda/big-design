StatefulTree

The StatefulTree component is used to display a tree of items. Useful for defining a tree of categories or subcollections.

function Example() {
  const nodes = [
    {
      id: '0',
      value: 0,
      label: 'Category',
      children: [
        {
          id: '5',
          value: 5,
          label: 'Category',
          children: [{ id: '9', value: 9, label: 'Category' }],
        },
      ],
    },
    {
      id: '1',
      value: 1,
      label: 'Category',
      children: [{ id: '6', value: 6, label: 'Category' }],
    },
    { id: '2', value: 2, label: 'Category' },
    {
      id: '3',
      value: 3,
      label: 'Category',
      children: [{ id: '7', value: 7, label: 'Category' }],
    },
    { id: '4', value: 4, label: 'Category', children: [{ id: '8', value: 8, label: 'Category' }] },
  ];
  return (
    <StatefulTree
      defaultExpanded={['0', '5', '1']}
      defaultSelected={['9', '3', '7']}
      disabledNodes={['1']}
      nodes={nodes}
      selectable="multi"
    />
  );
}

function Example() {
  const nodes = [
    {
      id: '0',
      value: 0,
      label: 'Category',
      children: [{ id: '3', value: 3, label: 'Subcategory' }],
    },
    { id: '1', value: 1, label: 'Category' },
    { id: '2', value: 2, label: 'Category' },
  ];
  return <StatefulTree selectable="radio" nodes={nodes} defaultExpanded={['0']} disabledNodes={['0', '2']} />;
}

Custom Icons

You can replace the folder icon with a custom icon of your choice.

If you use the iconless prop, ALL icons will be hidden (including custom ones).
  • Storefront - US
    • Subcategory
  • Storefront - CA
  • Storefront - EU

Edit the code below to see your changes live!

function Example() {
  const nodes = [
    {
      id: '0',
      icon: <StoreIcon color="primary" />,
      label: 'Storefront - US',
      children: [{ id: '3', label: 'Subcategory' }],
    },
    { id: '1', icon: <LanguageIcon color="primary" />, label: 'Storefront - CA' },
    { id: '2', icon: <AssignmentIcon color="primary" />, label: 'Storefront - EU' },
  ];
  return <StatefulTree defaultExpanded={['0']} nodes={nodes} />;
}
